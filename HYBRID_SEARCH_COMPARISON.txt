  4. Golf Putter Pro                    (semantic:0.78, keyword:1.0, final:0.87)
     ↑ "putter" in name = 1.0
     
  5. Golf Wedge Set                     (semantic:0.76, keyword:1.0, final:0.86)
     ↑ "wedge" in name = 1.0
     
  6. Golf 7 Wood                        (semantic:0.74, keyword:1.0, final:0.84)
     ↑ "wood" in name = 1.0
     
  7. Golf Hybrid Club                   (semantic:0.72, keyword:1.0, final:0.83)
     ↑ "club" in name = 1.0
